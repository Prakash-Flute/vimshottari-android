        // Initialize
        document.addEventListener('DOMContentLoaded', () => {
            initMonaco();

        // Monaco Editor Initialization
        function initMonaco() {
            require.config({ paths: { 'vs': 'https://cdnjs.cloudflare.com/ajax/libs/monaco-editor/0.45.0/min/vs' }});
            require(['vs/editor/editor.main'], function() {
                state.monaco = monaco;

                    value: '',
                    language: 'plaintext',

        // File Operations
        async function loadLocalFiles(path) {
            showLoading('local', true);
            
            // Simulate API call - Replace with actual fetch to your backend
            setTimeout(() => {
                const files = mockLocalFiles[path] || generateMockFiles(path);
                state.local.files = files;
                state.local.currentPath = path;
                renderFileList('local', files);
                updateBreadcrumb('local', path);
                document.getElementById('local-count').textContent = `${files.length} items`;
                document.getElementById('local-path').textContent = path;
                showLoading('local', false);
            }, 200);
        }

        async function loadGitHubFiles(path) {
            if (!state.github.token || !state.github.repo) {
                document.getElementById('github-files').innerHTML = `
                    <div class="text-center text-gray-500 mt-8 text-sm">
                        <i class="fas fa-key text-4xl mb-2 opacity-50"></i>
                        <p>Please configure GitHub token and repository</p>
                    </div>
                `;
                return;
            }
            
            showLoading('github', true);
            
            // GitHub API Implementation
            try {
                const response = await fetch(`https://api.github.com/repos/${state.github.repo}/contents/${path}?ref=${state.github.branch}`, {
                    headers: {
                        'Authorization': `token ${state.github.token}`,
                        'Accept': 'application/vnd.github.v3+json'
                    }
                });
                
                if (!response.ok) throw new Error('Failed to fetch');
                
                const data = await response.json();
                const files = Array.isArray(data) ? data : [data];
                
                state.github.files = files.map(f => ({
                    name: f.name,
                    type: f.type === 'dir' ? 'directory' : 'file',
                    size: f.size || 0,
                    sha: f.sha,
                    path: f.path,
                    download_url: f.download_url
                }));
                
                state.github.currentPath = path;
                renderFileList('github', state.github.files);
                updateBreadcrumb('github', path);
                document.getElementById('github-count').textContent = `${state.github.files.length} items`;
            } catch (err) {
                showNotification('Error loading GitHub: ' + err.message, 'error');
            } finally {
                showLoading('github', false);
            }
        }

        function renderFileList(panel, files) {
            const container = document.getElementById(`${panel}-files`);
            container.innerHTML = '';
            
            files.forEach((file, index) => {
                const item = document.createElement('div');
                item.className = 'file-item flex items-center gap-3 px-3 py-2 rounded cursor-pointer select-none';
                item.draggable = true;
                item.dataset.name = file.name;
                item.dataset.type = file.type;
                item.dataset.index = index;
                
                const icon = getFileIcon(file.name, file.type);
                const size = file.type === 'directory' ? '--' : formatSize(file.size);
                
                item.innerHTML = `
                    <i class="${icon} text-lg ${file.type === 'directory' ? 'text-yellow-500' : 'text-blue-400'} w-6 text-center"></i>
                    <div class="flex-1 min-w-0">
                        <div class="text-sm font-medium truncate">${file.name}</div>
                        <div class="text-xs text-gray-500">${size} • ${file.modified || 'Unknown'}</div>
                    </div>
                `;
                
                // Events
                item.addEventListener('click', (e) => selectFile(panel, index, e));
                item.addEventListener('dblclick', () => openFile(panel, file));
                item.addEventListener('contextmenu', (e) => showContextMenu(e, panel, file));
                
                // Drag events
                item.addEventListener('dragstart', (e) => handleDragStart(e, panel, file));
                item.addEventListener('dragend', (e) => handleDragEnd(e));
                
                container.appendChild(item);
            });
        }

        // UI Helpers
        function getFileIcon(name, type) {
            if (type === 'directory') return 'fas fa-folder';
            
            const ext = name.split('.').pop().toLowerCase();
            const iconMap = {
                'js': 'fab fa-js text-yellow-400',
                'html': 'fab fa-html5 text-orange-500',
                'css': 'fab fa-css3 text-blue-500',
                'json': 'fas fa-code text-gray-400',
                'md': 'fas fa-markdown text-white',
                'txt': 'fas fa-file-alt text-gray-400',
                'py': 'fab fa-python text-blue-300',
                'jpg': 'fas fa-image text-purple-400',
                'png': 'fas fa-image text-purple-400'
            };
            return iconMap[ext] || 'fas fa-file text-gray-400';
        }

        function formatSize(bytes) {
            if (bytes === 0) return '0 B';
            const k = 1024;
            const sizes = ['B', 'KB', 'MB', 'GB'];
            const i = Math.floor(Math.log(bytes) / Math.log(k));
            return parseFloat((bytes / Math.pow(k, i)).toFixed(1)) + ' ' + sizes[i];
        }

        function updateBreadcrumb(panel, path) {
            const container = document.getElementById(`${panel}-breadcrumb`);
            container.innerHTML = '';
            
            const parts = path.split('/').filter(p => p);
            let currentPath = '';
            
            // Root
            const root = document.createElement('span');
            root.className = 'breadcrumb-item cursor-pointer hover:text-blue-400';
            root.textContent = panel === 'local' ? '/' : 'root';
            root.onclick = () => navigateTo(panel, '');
            container.appendChild(root);
            
            parts.forEach((part, i) => {
                const separator = document.createElement('span');
                separator.className = 'text-gray-600';
                separator.textContent = ' / ';
                container.appendChild(separator);
                
                currentPath += (panel === 'local' ? '/' : '') + part;
                const span = document.createElement('span');
                span.className = 'breadcrumb-item cursor-pointer hover:text-blue-400';
                span.textContent = part;
                const pathSnapshot = currentPath;
                span.onclick = () => navigateTo(panel, pathSnapshot);
                container.appendChild(span);
            });
        }

        // Navigation
        function navigateTo(panel, path) {
            if (panel === 'local') {
                loadLocalFiles(path || '/storage/emulated/0');
            } else {
                loadGitHubFiles(path);
            }
        }

        function navigateUp(panel) {
            const current = panel === 'local' ? state.local.currentPath : state.github.currentPath;
            if (!current || current === '/storage/emulated/0') return;
            
            const parent = current.split('/').slice(0, -1).join('/') || '/storage/emulated/0';
            navigateTo(panel, parent);
        }

        function refreshPanel(panel) {
            if (panel === 'local') {
                loadLocalFiles(state.local.currentPath);
            } else {
                loadGitHubFiles(state.github.currentPath);
            }
        }

        // File Selection & Opening
        function selectFile(panel, index, event) {
            const container = document.getElementById(`${panel}-files`);
            const items = container.querySelectorAll('.file-item');
            
            if (!event.ctrlKey && !event.shiftKey) {
                items.forEach(i => i.classList.remove('selected'));
            }
            
            items[index].classList.add('selected');
            
            if (panel === 'local') {
                state.local.selected = state.local.files[index];
            } else {
                state.github.selected = state.github.files[index];
            }
        }

        async function openFile(panel, file) {
            if (file.type === 'directory') {
                const newPath = panel === 'local' 
                    ? `${state.local.currentPath}/${file.name}`
                    : `${state.github.currentPath}/${file.name}`.replace(/^\/+/, '');
                navigateTo(panel, newPath);
                return;
            }
            
            // Open in editor
            openEditor(panel, file);
        }

        // Editor Operations
        async function openEditor(panel, file) {
            state.editor.currentFile = { panel, file };
            state.editor.visible = true;
            
            document.getElementById('editor-filename').textContent = file.name;

```javascript
const express = require('express');
const router = express.Router();
const filesystemService = require('../services/filesystemService');
const githubService = require('../services/githubService');

// POST /api/transfer/local-to-github
router.post('/local-to-github', async (req, res) => {
    try {
        const { localPath, githubPath, message } = req.body;
        
        if (!localPath || !githubPath) {
            return res.status(400).json({
                success: false,
                error: 'Both localPath and githubPath required'
            });
        }

        if (!githubService.isConnected()) {
            return res.status(401).json({
                success: false,
                error: 'GitHub not connected'
            });
        }

        // Step 1: Read local file
        console.log(`[Transfer] Reading local file: ${localPath}`);
        const content = await filesystemService.readFile(localPath);
        
        // Step 2: Check if file exists on GitHub (for update vs create)
        let sha = null;
        try {
            const existingFile = await githubService.getFileInfo(githubPath);
            sha = existingFile.sha;
        } catch (err) {
            // File doesn't exist, will create new
        }

        // Step 3: Transfer to GitHub
        console.log(`[Transfer] Writing to GitHub: ${githubPath}`);
        let result;
        const commitMessage = message || `Transfer ${localPath.split('/').pop()} via MT-Git Manager`;
        
        if (sha) {
            // Update existing
            result = await githubService.updateFile(githubPath, content, sha, commitMessage);
        } else {
            // Create new
            result = await githubService.createFile(githubPath, content, commitMessage);
        }

        res.json({
            success: true,
            message: 'Transfer completed successfully',
            source: localPath,
            destination: githubPath,
            commit: result.commit
        });
    } catch (error) {
        console.error('[Transfer Error]', error);
        res.status(500).json({
            success: false,
            error: error.message,
            stage: error.stage || 'unknown'
        });
    }
});

router.post('/github-to-local', async (req, res) => {

// POST /api/transfer/batch
router.post('/batch', async (req, res) => {
    try {
        const { operations } = req.body;
        
        if (!Array.isArray(operations)) {
            return res.status(400).json({
                success: false,
                error: 'Operations array required'
            });
        }

        const results = [];
        const errors = [];

        for (const op of operations) {
            try {
                if (op.direction === 'local-to-github') {
                    const content = await filesystemService.readFile(op.source);
                    let sha = null;
                    try {
                        const info = await githubService.getFileInfo(op.destination);
                        sha = info.sha;
                    } catch (e) {}
                    
                    if (sha) {
                        await githubService.updateFile(op.destination, content, sha, op.message);
                    } else {
                        await githubService.createFile(op.destination, content, op.message);
                    }
                } else {
                    const content = await githubService.getFileContent(op.source);
                    await filesystemService.writeFile(op.destination, content);
                }
                results.push({ success: true, operation: op });
            } catch (err) {
                errors.push({ success: false, operation: op, error: err.message });
            }
        }

        res.json({
            success: errors.length === 0,
            completed: results.length,
            failed: errors.length,
            results,
            errors
        });
    } catch (error) {
        res.status(500).json({
            success: false,
            error: error.message
        });
    }
});
```

```javascript
// server/services/filesystemService.js
const fs = require('fs').promises;
const path = require('path');
const pathUtils = require('../utils/pathUtils');

const BASE_PATH = '/storage/emulated/0';

class FilesystemService {
    constructor() {
        this.basePath = BASE_PATH;
    }

    // Ensure path is within allowed directory
    sanitizePath(inputPath) {
        // Normalize path
        let normalized = path.normalize(inputPath);
        
        // Remove any .. sequences
        if (normalized.includes('..')) {
            throw new Error('Invalid path: directory traversal detected');
        }
        
        // Ensure path starts with base path
        if (!normalized.startsWith(this.basePath)) {
            // If relative path provided, join with base
            if (!normalized.startsWith('/')) {
                normalized = path.join(this.basePath, normalized);
            } else {
                throw new Error('Invalid path: outside allowed directory');
            }
        }
        
        return normalized;
    }

    async listDirectory(dirPath) {
        const safePath = this.sanitizePath(dirPath);
        
        try {
            const entries = await fs.readdir(safePath, { withFileTypes: true });
            
            const files = await Promise.all(entries.map(async (entry) => {
                const fullPath = path.join(safePath, entry.name);
                const stats = await fs.stat(fullPath).catch(() => null);
                
                return {
                    name: entry.name,
                    type: entry.isDirectory() ? 'directory' : 'file',
                    size: stats ? stats.size : 0,
                    modified: stats ? stats.mtime.toISOString() : new Date().toISOString(),
                    permissions: stats ? stats.mode.toString(8) : 'unknown'
                };
            }));
            
            // Sort: directories first, then alphabetically
            return files.sort((a, b) => {
                if (a.type === b.type) return a.name.localeCompare(b.name);
                return a.type === 'directory' ? -1 : 1;
            });
        } catch (error) {
            throw new Error(`Cannot read directory: ${error.message}`);
        }
    }

    async readFile(filePath) {
        const safePath = this.sanitizePath(filePath);
        
        try {
            // Check if file is binary (images, etc.)
            const ext = path.extname(safePath).toLowerCase();
            const binaryExts = ['.png', '.jpg', '.jpeg', '.gif', '.ico', '.pdf', '.zip'];
            
            if (binaryExts.includes(ext)) {
                return await fs.readFile(safePath);
            }
            
            return await fs.readFile(safePath, 'utf-8');
        } catch (error) {
            throw new Error(`Cannot read file: ${error.message}`);
        }
    }

    async writeFile(filePath, content) {
        const safePath = this.sanitizePath(filePath);
        
        try {
            // Ensure directory exists
            const dir = path.dirname(safePath);
            await fs.mkdir(dir, { recursive: true });
            
            await fs.writeFile(safePath, content, 'utf-8');
            return true;
        } catch (error) {
            throw new Error(`Cannot write file: ${error.message}`);
        }
    }

    async deleteFile(filePath) {
        const safePath = this.sanitizePath(filePath);
        
        try {
            const stats = await fs.stat(safePath);
            
            if (stats.isDirectory()) {
                await fs.rmdir(safePath, { recursive: true });
            } else {
                await fs.unlink(safePath);
            }
            
            return true;
        } catch (error) {
            throw new Error(`Cannot delete: ${error.message}`);
        }
    }

    async renameFile(oldPath, newPath) {
        const safeOld = this.sanitizePath(oldPath);
        const safeNew = this.sanitizePath(newPath);
        
        try {
            await fs.rename(safeOld, safeNew);
            return true;
        } catch (error) {
            throw new Error(`Cannot rename: ${error.message}`);
        }
    }

    async createDirectory(dirPath) {
        const safePath = this.sanitizePath(dirPath);
        
        try {
            await fs.mkdir(safePath, { recursive: true });
            return true;
        } catch (error) {
            throw new Error(`Cannot create directory: ${error.message}`);
        }
    }

    async fileExists(filePath) {
        try {
            const safePath = this.sanitizePath(filePath);
            await fs.access(safePath);
            return true;
        } catch {
            return false;
        }
    }

    async getFileStats(filePath) {
        const safePath = this.sanitizePath(filePath);
        return await fs.stat(safePath);
    }
}

module.exports = new FilesystemService();
```

```javascript
// server/services/githubService.js
const https = require('https');

class GitHubService {
    constructor() {
        this.token = null;
        this.repo = null;
        this.owner = null;
        this.repoName = null;
        this.apiBase = 'api.github.com';
    }

    setCredentials(token, repo) {
        this.token = token;
        this.repo = repo;
        
        // Parse owner/repo format
        const parts = repo.split('/');
        if (parts.length !== 2) {
            throw new Error('Repository must be in format "owner/repo"');
        }
        this.owner = parts[0];
        this.repoName = parts[1];
    }

    isConnected() {
        return !!(this.token && this.repo && this.owner && this.repoName);
    }

    // Make authenticated request to GitHub API
    async request(path, method = 'GET', data = null) {
        return new Promise((resolve, reject) => {
            const options = {
                hostname: this.apiBase,
                path: path,
                method: method,
                headers: {
                    'Authorization': `Bearer ${this.token}`,
                    'Accept': 'application/vnd.github+json',
                    'User-Agent': 'MT-Git-Manager/1.0',
                    'X-GitHub-Api-Version': '2022-11-28'
                }
            };

            if (data) {
                options.headers['Content-Type'] = 'application/json';
            }

            const req = https.request(options, (res) => {
                let body = '';
                res.on('data', chunk => body += chunk);
                res.on('end', () => {
                    try {
                        const parsed = JSON.parse(body);
                        if (res.statusCode >= 200 && res.statusCode < 300) {
                            resolve(parsed);
                        } else {
                            reject(new Error(parsed.message || `HTTP ${res.statusCode}`));
                        }
                    } catch (e) {
                        reject(new Error('Invalid JSON response'));
                    }
                });
            });

            req.on('error', (err) => reject(err));

            if (data) {
                req.write(JSON.stringify(data));
            }
            
            req.end();
        });
    }

    async validateConnection(token, repo) {
        // Temporarily set credentials for validation
        const prevToken = this.token;
        const prevRepo = this.repo;
        
        this.token = token;
        this.repo = repo;
        const parts = repo.split('/');
        this.owner = parts[0];
        this.repoName = parts[1];
        
        try {
            const result = await this.request(`/repos/${this.owner}/${this.repoName}`);
            return result;
        } catch (error) {
            // Restore previous credentials on failure
            this.token = prevToken;
            this.repo = prevRepo;
            this.owner = null;
            this.repoName = null;
            throw error;
        }
    }

    async listContents(path = '') {
        const url = `/repos/${this.owner}/${this.repoName}/contents/${path}?ref=main`;
        const data = await this.request(url);
        
        // Handle both array (directory) and object (single file) responses
        const items = Array.isArray(data) ? data : [data];
        
        return items.map(item => ({
            name: item.name,
            type: item.type === 'dir' ? 'directory' : 'file',
            path: item.path,
            sha: item.sha,
            size: item.size || 0,
            download_url: item.download_url,
            html_url: item.html_url
        }));
    }

    async getFileContent(path) {
        const url = `/repos/${this.owner}/${this.repoName}/contents/${path}?ref=main`;
        const data = await this.request(url);
        
        if (data.content) {
            // Decode base64 content
            return Buffer.from(data.content, 'base64').toString('utf-8');
        }
        
        throw new Error('No content in response');
    }

    async getFileInfo(path) {
        const url = `/repos/${this.owner}/${this.repoName}/contents/${path}?ref=main`;
        return await this.request(url);
    }

    async createFile(path, content, message) {
        const url = `/repos/${this.owner}/${this.repoName}/contents/${path}`;
        
        // Encode content to base64
        const encodedContent = Buffer.from(content).toString('base64');
        
        const data = {
            message: message,
            content: encodedContent,
            branch: 'main'
        };

        return await this.request(url, 'PUT', data);
    }

    async updateFile(path, content, sha, message) {
        const url = `/repos/${this.owner}/${this.repoName}/contents/${path}`;
        
        const encodedContent = Buffer.from(content).toString('base64');
        
        const data = {
            message: message,
            content: encodedContent,
            sha: sha,
            branch: 'main'
        };

        return await this.request(url, 'PUT', data);
    }

    async deleteFile(path, sha, message) {
        const url = `/repos/${this.owner}/${this.repoName}/contents/${path}`;
        
        const data = {
            message: message,
            sha: sha,
            branch: 'main'
        };

        return await this.request(url, 'DELETE', data);
    }

    async listBranches() {
        const url = `/repos/${this.owner}/${this.repoName}/branches`;
        return await this.request(url);
    }
}

module.exports = new GitHubService();
```

```javascript
        const textTypes = ['.txt', '.md', '.json', '.js', '.ts', '.html', '.css', 
                          '.xml', '.yaml', '.yml', '.py', '.java', '.c', '.cpp',
                          '.go', '.rs', '.php', '.rb', '.swift', '.kt'];

        return textTypes.includes(ext);
    },

    isImageFile(filename) {
        const imageTypes = ['.png', '.jpg', '.jpeg', '.gif', '.svg', '.ico', '.bmp', '.webp'];
        const ext = path.extname(filename).toLowerCase();
        return imageTypes.includes(ext);
    },

    formatFileSize(bytes) {
        if (bytes === 0) return '0 B';
        const k = 1024;
        const sizes = ['B', 'KB', 'MB', 'GB', 'TB'];
        const i = Math.floor(Math.log(bytes) / Math.log(k));
        return parseFloat((bytes / Math.pow(k, i)).toFixed(2)) + ' ' + sizes[i];
    },

        const ext = path.extname(filename).toLowerCase();

            '.java': 'java',
            '.png': 'image',
            '.jpg': 'image',
            '.pdf': 'pdf',
            '.zip': 'zip'
        };
        
        return iconMap[ext] || 'file';
```

```html
        // GitHub API
        async function connectGitHub(silent = false) {
            const token = document.getElementById('gh-token').value;
            const repo = document.getElementById('gh-repo').value;
            
            if (!token || !repo) {
                if (!silent) showNotification('Please enter token and repository', 'error');
                return;
            }
            
            try {
                const response = await fetch(`${API_BASE}/github/connect`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ token, repo })
                });
                
                const data = await response.json();
                
                if (data.success) {
                    state.github.connected = true;
                    state.github.token = token;
                    state.github.repo = repo;
                    localStorage.setItem('gh_token', token);
                    localStorage.setItem('gh_repo', repo);
                    
                    document.getElementById('github-config').classList.add('hidden');
                    showNotification('Connected to GitHub', 'success');
                    await loadGitHubFiles('');
                } else {
                    throw new Error(data.error);
                }
            } catch (error) {
                showNotification(`GitHub connection failed: ${error.message}`, 'error');
                state.github.connected = false;
            }
        }

        // Transfer API
        async function transferFile(from, to, file) {
            if (file.type === 'directory') {
                showNotification('Directory transfer not yet implemented', 'warning');
                return;
            }
            
            showTransferProgress(true, `Transferring ${file.name}...`);
            
            try {
                if (from === 'local' && to === 'github') {
                    const localFilePath = `${state.local.currentPath}/${file.name}`;
                    const githubFilePath = state.github.currentPath ? 
                        `${state.github.currentPath}/${file.name}` : file.name;
                    
                    const response = await fetch(`${API_BASE}/transfer/local-to-github`, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify({
                            localPath: localFilePath,
                            githubPath: githubFilePath
                        })
                    });
                    
                    const data = await response.json();
                    if (!data.success) throw new Error(data.error);
                    
                } else if (from === 'github' && to === 'local') {
                    const githubFilePath = file.path || file.name;
                    const localFilePath = `${state.local.currentPath}/${file.name}`;
                    
                    const response = await fetch(`${API_BASE}/transfer/github-to-local`, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify({
                            githubPath: githubFilePath,
                            localPath: localFilePath
                        })
                    });
                    
                    const data = await response.json();
                    if (!data.success) throw new Error(data.error);
                }
                
                showTransferProgress(false);
                showNotification('Transfer complete', 'success');
                refreshPanel(to);
            } catch (error) {
                showTransferProgress(false);
                showNotification(`Transfer failed: ${error.message}`, 'error');
            }
        }

        // UI Rendering
        function renderFileList(panel, files) {
            const container = document.getElementById(`${panel}-files`);
            container.innerHTML = '';
            
            if (files.length === 0) {
                container.innerHTML = `
                    <div class="text-center text-gray-500 mt-8 text-sm">
                        <i class="fas fa-folder-open text-4xl mb-2 opacity-30"></i>
                        <p>Empty folder</p>
                    </div>
                `;
                return;
            }
            
            files.forEach((file, index) => {
                const item = document.createElement('div');
                item.className = 'file-item flex items-center gap-3 px-3 py-2 rounded cursor-pointer select-none';
                item.draggable = true;
                item.dataset.name = file.name;
                item.dataset.type = file.type;
                item.dataset.index = index;
                
                const icon = getFileIcon(file.name, file.type);
                const size = file.type === 'directory' ? '--' : formatSize(file.size);
                
                item.innerHTML = `
                    <i class="${icon} text-lg ${file.type === 'directory' ? 'text-yellow-500' : 'text-blue-400'} w-6 text-center"></i>
                    <div class="flex-1 min-w-0">
                        <div class="text-sm font-medium truncate">${file.name}</div>
                        <div class="text-xs text-gray-500">${size} • ${formatDate(file.modified)}</div>
                    </div>
                `;
                
                item.addEventListener('click', (e) => selectFile(panel, index, e));
                item.addEventListener('dblclick', () => openFile(panel, file));
                item.addEventListener('contextmenu', (e) => showContextMenu(e, panel, file));
                item.addEventListener('dragstart', (e) => handleDragStart(e, panel, file));
                item.addEventListener('dragend', (e) => handleDragEnd(e));
                
                container.appendChild(item);
            });
        }

        function getFileIcon(name, type) {
            if (type === 'directory') return 'fas fa-folder';
            
            const ext = name.split('.').pop().toLowerCase();
```

```javascript
// server/services/transferService.js
const filesystemService = require('./filesystemService');
const githubService = require('./githubService');
const path = require('path');

/**
 * Service for handling complex transfer operations between local and GitHub
 * Supports recursive transfers, batch operations, and progress tracking
 */
class TransferService {
    constructor() {
        this.activeTransfers = new Map();
        this.transferId = 0;
    }

    /**
     * Recursively transfer a local folder to GitHub
     * @param {string} localPath - Source folder path
     * @param {string} githubPath - Destination path in repo
     * @param {Function} progressCallback - Called with progress updates
     * @returns {Promise<Object>} Transfer result with stats
     */
    async transferFolderLocalToGithub(localPath, githubPath, progressCallback = () => {}) {
        const transferId = ++this.transferId;
        const stats = { files: 0, folders: 0, errors: [], completed: [] };
        
        this.activeTransfers.set(transferId, { status: 'scanning', progress: 0 });
        
        try {
            // Phase 1: Scan directory structure
            const fileList = await this.scanLocalDirectory(localPath);
            const totalItems = fileList.length;
            
            this.activeTransfers.set(transferId, { status: 'uploading', progress: 0, total: totalItems });
            
            // Phase 2: Upload in batches (GitHub recommends max 100 files per commit)
            const batchSize = 50;
            for (let i = 0; i < fileList.length; i += batchSize) {
                const batch = fileList.slice(i, i + batchSize);
                await this.uploadBatchToGithub(batch, localPath, githubPath, stats);
                
                const progress = Math.round(((i + batch.length) / totalItems) * 100);
                this.activeTransfers.set(transferId, { status: 'uploading', progress, total: totalItems });
                progressCallback({ status: 'uploading', progress, current: i + batch.length, total: totalItems });
            }
            
            this.activeTransfers.set(transferId, { status: 'completed', progress: 100 });
            return { success: true, transferId, stats };
            
        } catch (error) {
            this.activeTransfers.set(transferId, { status: 'error', error: error.message });
            throw error;
        }
    }

    /**
     * Recursively transfer GitHub folder to local storage
     * @param {string} githubPath - Source folder in repo
     * @param {string} localPath - Destination local path
     * @param {Function} progressCallback - Progress updates
     */
    async transferFolderGithubToLocal(githubPath, localPath, progressCallback = () => {}) {
        const transferId = ++this.transferId;
        const stats = { files: 0, folders: 0, errors: [], completed: [] };
        
        this.activeTransfers.set(transferId, { status: 'scanning', progress: 0 });
        
        try {
            // Phase 1: Get tree recursively
            const tree = await this.getGithubTreeRecursive(githubPath);
            const files = tree.filter(item => item.type === 'blob');
            const totalItems = files.length;
            
            this.activeTransfers.set(transferId, { status: 'downloading', progress: 0, total: totalItems });
            
            // Phase 2: Download with concurrency limit
            const concurrency = 5;
            for (let i = 0; i < files.length; i += concurrency) {
                const batch = files.slice(i, i + concurrency);
                await Promise.all(batch.map(file => this.downloadGithubFile(file, localPath, githubPath, stats)));
                
                const progress = Math.round(((i + batch.length) / totalItems) * 100);
                this.activeTransfers.set(transferId, { status: 'downloading', progress, total: totalItems });
                progressCallback({ status: 'downloading', progress, current: i + batch.length, total: totalItems });
            }
            
            this.activeTransfers.set(transferId, { status: 'completed', progress: 100 });
            return { success: true, transferId, stats };
            
        } catch (error) {
            this.activeTransfers.set(transferId, { status: 'error', error: error.message });
            throw error;
        }
    }

    /**
     * Scan local directory recursively
     * @param {string} dirPath - Root directory to scan
     * @param {string} [basePath] - Internal use for recursion
     * @returns {Promise<Array>} Flat list of files with relative paths
     */
    async scanLocalDirectory(dirPath, basePath = '') {
        const files = [];
        const entries = await filesystemService.listDirectory(dirPath);
        
        for (const entry of entries) {
            const fullPath = path.join(dirPath, entry.name);
            const relativePath = path.join(basePath, entry.name);
            
            if (entry.type === 'directory') {
                const subFiles = await this.scanLocalDirectory(fullPath, relativePath);
                files.push(...subFiles);
            } else {
                files.push({
                    localPath: fullPath,
                    relativePath: relativePath,
                    name: entry.name,
                    size: entry.size
                });
            }
        }
        
        return files;
    }

    /**
     * Upload a batch of files to GitHub
     * @private
     */
    async uploadBatchToGithub(files, localBasePath, githubBasePath, stats) {
        for (const file of files) {
            try {
                const content = await filesystemService.readFile(file.localPath);
                const githubPath = path.join(githubBasePath, file.relativePath).replace(/\\/g, '/');
                
                // Check if file exists
                let sha = null;
                try {
                    const existing = await githubService.getFileInfo(githubPath);
                    sha = existing.sha;
                } catch (e) {
                    // File doesn't exist, will create
                }
                
                if (sha) {
                    await githubService.updateFile(githubPath, content, sha, `Update ${file.relativePath} via MT-Git Manager`);
                } else {
                    await githubService.createFile(githubPath, content, `Create ${file.relativePath} via MT-Git Manager`);
                }
                
                stats.completed.push(file.relativePath);
                stats.files++;
            } catch (error) {
                stats.errors.push({ file: file.relativePath, error: error.message });
            }
        }
    }

    /**
     * Get recursive tree from GitHub
     * @private
     */
    async getGithubTreeRecursive(path) {
        // Use Git Trees API for recursive fetching
        const { owner, repoName } = githubService;
        const treeSha = await githubService.getBranchTreeSha('main');
        
        const response = await githubService.request(
            `/repos/${owner}/${repoName}/git/trees/${treeSha}?recursive=1`
        );
        
        // Filter by path prefix if specified
        if (path) {
            return response.tree.filter(item => item.path.startsWith(path));
        }
        return response.tree;
    }

    /**
     * Download single file from GitHub to local
     * @private
     */
    async downloadGithubFile(file, localBasePath, githubBasePath, stats) {
        try {
            const relativePath = file.path.replace(githubBasePath + '/', '');
            const localFilePath = path.join(localBasePath, relativePath);
            
            // Ensure directory exists
            const dir = path.dirname(localFilePath);
            await filesystemService.createDirectory(dir);
            
            // Get content
            const content = await githubService.getFileContent(file.path);
            await filesystemService.writeFile(localFilePath, content);
            
            stats.completed.push(relativePath);
            stats.files++;
        } catch (error) {
            stats.errors.push({ file: file.path, error: error.message });
        }
    }

    getTransferStatus(transferId) {
        return this.activeTransfers.get(transferId);
    }
}
```

```javascript
class SelectionManager {
    constructor() {
        this.selections = {
            local: new Set(),
            github: new Set()

        this.lastSelected = {
            local: null,
            github: null
        };
        this.anchorIndex = {
            local: null,
            github: null
        };
    }

    /**
     * Handle click with modifier keys
     * @param {string} panel - 'local' or 'github'
     * @param {number} index - File index
     * @param {Object} file - File object
     * @param {MouseEvent} event - Click event
     * @returns {Set} Current selection
     */
    handleClick(panel, index, file, event) {
        const currentSelection = this.selections[panel];
        
        if (event.ctrlKey || event.metaKey) {
            // Toggle selection
            if (currentSelection.has(index)) {
                currentSelection.delete(index);
            } else {
                currentSelection.add(index);
                this.anchorIndex[panel] = index;
            }
        } else if (event.shiftKey && this.anchorIndex[panel] !== null) {
            // Range selection
            const start = Math.min(this.anchorIndex[panel], index);
            const end = Math.max(this.anchorIndex[panel], index);
            
            currentSelection.clear();
            for (let i = start; i <= end; i++) {
                currentSelection.add(i);
            }
        } else {
            // Single selection
            currentSelection.clear();
            currentSelection.add(index);
            this.anchorIndex[panel] = index;
        }
        
        this.lastSelected[panel] = file;
        return new Set(currentSelection);
    }

    /**
     * Select all files in panel
     */
    selectAll(panel, fileCount) {
        this.selections[panel].clear();
        for (let i = 0; i < fileCount; i++) {
            this.selections[panel].add(i);
        }
        return new Set(this.selections[panel]);
    }

    /**
     * Clear selection
     */
    clearSelection(panel) {
        this.selections[panel].clear();
        this.anchorIndex[panel] = null;
        return new Set();
    }

    /**
     * Get selected files from file list
     */
    getSelectedFiles(panel, files) {
        return Array.from(this.selections[panel]).map(index => ({
            index,
            file: files[index]
        })).filter(item => item.file !== undefined);
    }

    /**
     * Check if index is selected
     */
    isSelected(panel, index) {
        return this.selections[panel].has(index);
    }

    /**
     * Get selection count
     */
    getCount(panel) {
        return this.selections[panel].size;
    }
}
```

```javascript
// server/services/transferQueue.js
const EventEmitter = require('events');

/**
 * High-performance transfer queue with workers, retries, and progress tracking
 */
class TransferQueue extends EventEmitter {
    constructor(options = {}) {
        super();
        this.concurrency = options.concurrency || 3;
        this.retryAttempts = options.retryAttempts || 3;
        this.retryDelay = options.retryDelay || 1000;
        this.queue = [];
        this.active = new Map();
        this.completed = [];
        this.failed = [];
        this.stats = { transferred: 0, failed: 0, total: 0 };
        this.running = false;
        this.paused = false;
    }

    /**
     * Add transfer job to queue
     * @param {Object} job - Transfer job definition
     * @param {string} job.type - 'local-to-github' or 'github-to-local'
     * @param {string} job.source - Source path
     * @param {string} job.destination - Destination path
     * @param {Object} job.meta - Additional metadata (size, type, etc)
     */
    add(job) {
        const jobId = `job_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`;
        const jobWithId = { 
            id: jobId, 
            ...job, 
            status: 'queued',
            attempts: 0,
            createdAt: Date.now()
        };
        
        this.queue.push(jobWithId);
        this.stats.total++;
        this.emit('jobAdded', jobWithId);
        
        if (!this.running) {
            this.start();
        }
        return jobId;
    }

    /**
     * Add multiple jobs (batch)
     */
    addBatch(jobs) {
        return jobs.map(job => this.add(job));
    }

    /**
     * Start processing queue
     */
    async start() {
        if (this.running || this.paused) return;
        this.running = true;
        this.emit('started');

        while (this.queue.length > 0 || this.active.size > 0) {
            if (this.paused) {
                await this.waitForResume();
                continue;
            }

            // Fill active slots
            while (this.active.size < this.concurrency && this.queue.length > 0) {
                const job = this.queue.shift();
                this.processJob(job);
            }

            // Wait for at least one job to complete if queue is full
            if (this.active.size >= this.concurrency || this.queue.length === 0) {
                await this.waitForCompletion();
            }
        }

        this.running = false;
        this.emit('completed', this.stats);
    }

    /**
     * Process individual job with retry logic
     */
    async processJob(job) {
        this.active.set(job.id, job);
        job.status = 'active';
        job.startedAt = Date.now();
        this.emit('jobStarted', job);

        try {
            const result = await this.executeTransfer(job);
            job.status = 'completed';
            job.completedAt = Date.now();
            this.completed.push(job);
            this.stats.transferred++;
            this.emit('jobCompleted', job, result);
        } catch (error) {
            job.attempts++;
            
            if (job.attempts < this.retryAttempts) {
                // Retry with exponential backoff
                job.status = 'retrying';
                this.emit('jobRetrying', job, error);
                await this.delay(this.retryDelay * Math.pow(2, job.attempts - 1));
                this.queue.unshift(job); // Put back in queue
            } else {
                job.status = 'failed';
                job.error = error.message;
                job.failedAt = Date.now();
                this.failed.push(job);
                this.stats.failed++;
                this.emit('jobFailed', job, error);
            }
        } finally {
            this.active.delete(job.id);
            this.emit('progress', this.getProgress());
        }
    }

    /**
     * Execute actual transfer based on job type
     */
    async executeTransfer(job) {
        const transferService = require('./transferService');
        
        if (job.type === 'local-to-github') {
            if (job.meta && job.meta.type === 'directory') {
                return await transferService.transferFolderLocalToGithub(
                    job.source, 
                    job.destination,
                    (progress) => this.emit('folderProgress', job.id, progress)
                );
            } else {
                // Single file transfer
                const content = await require('./filesystemService').readFile(job.source);
                let sha = null;
                try {
                    const existing = await require('./githubService').getFileInfo(job.destination);
                    sha = existing.sha;
                } catch (e) {}
                
                if (sha) {
                    return await require('./githubService').updateFile(
                        job.destination, content, sha, 
                        `Update ${job.meta.name} via MT-Git Manager`
                    );
                } else {
                    return await require('./githubService').createFile(
                        job.destination, content,
                        `Create ${job.meta.name} via MT-Git Manager`
                    );
                }
            }
        } else if (job.type === 'github-to-local') {
            if (job.meta && job.meta.type === 'directory') {
                return await transferService.transferFolderGithubToLocal(
                    job.source,
                    job.destination,
                    (progress) => this.emit('folderProgress', job.id, progress)
                );
            } else {
                const content = await require('./githubService').getFileContent(job.source);
                await require('./filesystemService').writeFile(job.destination, content);
                return { success: true };
            }
        }
        
        throw new Error(`Unknown transfer type: ${job.type}`);
    }

    /**
     * Pause queue processing
     */
    pause() {
        this.paused = true;
        this.emit('paused');
    }

    /**
     * Resume queue processing
     */
    resume() {
        this.paused = false;
        this.emit('resumed');
        if (!this.running) {
            this.start();
        }
    }

    /**
     * Cancel specific job or all jobs
     */
    cancel(jobId = null) {
        if (jobId) {
            const index = this.queue.findIndex(j => j.id === jobId);
            if (index > -1) {
                const job = this.queue.splice(index, 1)[0];
                job.status = 'cancelled';
                this.emit('jobCancelled', job);
            }
        } else {
            // Cancel all queued
            while (this.queue.length > 0) {
                const job = this.queue.shift();
                job.status = 'cancelled';
                this.emit('jobCancelled', job);
            }
        }
    }

    /**
     * Get current progress statistics
     */
    getProgress() {
        const total = this.stats.total;
        const completed = this.stats.transferred + this.stats.failed;
        const percent = total > 0 ? Math.round((completed / total) * 100) : 0;
        
        return {
            total,
            completed,
            failed: this.stats.failed,
            active: this.active.size,
            queued: this.queue.length,
            percent,
            isComplete: completed === total && this.queue.length === 0 && this.active.size === 0
        };
    }

    /**
     * Get detailed status report
     */
    getStatus() {
        return {
            running: this.running,
            paused: this.paused,
            progress: this.getProgress(),
            active: Array.from(this.active.values()),
            queued: this.queue.slice(0, 10), // First 10
            completed: this.completed.slice(-10), // Last 10
            failed: this.failed
        };
    }

    // Helper methods
    waitForCompletion() {
        return new Promise(resolve => {
            const check = () => {
                if (this.active.size < this.concurrency) {
                    resolve();
                } else {
                    setTimeout(check, 100);
                }
            };
            check();
        });
    }

    waitForResume() {
        return new Promise(resolve => {
            const check = () => {
                if (!this.paused) {
                    resolve();
                } else {
                    setTimeout(check, 100);
                }
            };
            check();
        });
    }

    delay(ms) {
        return new Promise(resolve => setTimeout(resolve, ms));
    }
}

module.exports = TransferQueue;
```

```javascript
// server/services/githubBatchService.js
const githubService = require('./githubService');

/**
 * Batch operations for GitHub API to minimize API calls
 * Groups multiple file changes into single commits where possible
 */
class GitHubBatchService {
    constructor() {
        this.pendingChanges = [];
        this.batchTimeout = null;
        this.batchDelay = 5000; // 5 seconds to collect changes
    }

    /**
     * Add file change to batch
     * @param {Object} change - Change operation
     * @param {string} change.operation - 'create', 'update', 'delete'
     * @param {string} change.path - File path
     * @param {string} change.content - File content (for create/update)
     * @param {string} change.sha - File SHA (for update/delete)
     */
    async queueChange(change) {
        this.pendingChanges.push({
            ...change,
            queuedAt: Date.now()
        });

        // Reset batch timer
        if (this.batchTimeout) {
            clearTimeout(this.batchTimeout);
        }

        // Auto-flush if batch is large
        if (this.pendingChanges.length >= 100) {
            await this.flushBatch();
        } else {
            // Set timer for auto-flush
            this.batchTimeout = setTimeout(() => this.flushBatch(), this.batchDelay);
        }

        return { queued: true, position: this.pendingChanges.length };
    }

    /**
     * Execute all pending changes as batch commit
     * Uses Git Trees API for atomic multi-file commits
     */
    async flushBatch() {
        if (this.pendingChanges.length === 0) return { success: true, changes: 0 };

        const changes = [...this.pendingChanges];
        this.pendingChanges = [];
        
        try {
            // Get current branch reference
            const ref = await githubService.request(
                `/repos/${githubService.owner}/${githubService.repoName}/git/ref/heads/${githubService.branch || 'main'}`
            );
            const currentCommit = ref.object.sha;

            // Get current tree
            const commit = await githubService.request(
                `/repos/${githubService.owner}/${githubService.repoName}/git/commits/${currentCommit}`
            );
            const baseTree = commit.tree.sha;

            // Build new tree with changes
            const tree = await this.buildTree(baseTree, changes);

            // Create commit
            const newCommit = await githubService.request(
                `/repos/${githubService.owner}/${githubService.repoName}/git/commits`,
                'POST',
                {
                    message: `MT-Git Manager batch update (${changes.length} files)`,
                    tree: tree.sha,
                    parents: [currentCommit]
                }
            );

            // Update branch reference
            await githubService.request(
                `/repos/${githubService.owner}/${githubService.repoName}/git/refs/heads/${githubService.branch || 'main'}`,
                'PATCH',
                { sha: newCommit.sha }
            );

            return {
                success: true,
                changes: changes.length,
                commit: newCommit.sha
            };

        } catch (error) {
            // Re-queue failed changes
            this.pendingChanges.unshift(...changes);
            throw error;
        }
    }

    /**
     * Build git tree from changes
     */
    async buildTree(baseTreeSha, changes) {
        // Get existing tree
        const baseTree = await githubService.request(
            `/repos/${githubService.owner}/${githubService.repoName}/git/trees/${baseTreeSha}?recursive=1`
        );

        // Create new tree items
        const treeItems = baseTree.tree.map(item => ({
            path: item.path,
            mode: item.mode,
            type: item.type,
            sha: item.sha
        }));

        // Apply changes
        for (const change of changes) {
            const index = treeItems.findIndex(item => item.path === change.path);
            
            if (change.operation === 'delete') {
                if (index > -1) {
                    treeItems.splice(index, 1);
                }
            } else {
                // Create or update - need to create blob first
                const blob = await githubService.request(
                    `/repos/${githubService.owner}/${githubService.repoName}/git/blobs`,
                    'POST',
                    {
                        content: Buffer.from(change.content).toString('base64'),
                        encoding: 'base64'
                    }
                );

                const newItem = {
                    path: change.path,
                    mode: '100644',
                    type: 'blob',
                    sha: blob.sha
                };

                if (index > -1) {
                    treeItems[index] = newItem;
                } else {
                    treeItems.push(newItem);
                }
            }
        }

        // Create new tree
        const newTree = await githubService.request(
            `/repos/${githubService.owner}/${githubService.repoName}/git/trees`,
            'POST',
            {
                base_tree: baseTreeSha,
                tree: treeItems.map(item => ({
                    path: item.path,
                    mode: item.mode,
                    type: item.type,
                    sha: item.sha
                }))
            }
        );

        return newTree;
    }

    /**
     * Immediate batch operation without queueing
     */
    async executeImmediate(changes) {
        this.pendingChanges = changes;
        return await this.flushBatch();
    }

    /**
     * Get pending changes count
     */
    getPendingCount() {
        return this.pendingChanges.length;
    }
}

module.exports = new GitHubBatchService();
```

```javascript
// server/services/syncEngine.js
const EventEmitter = require('events');
const filesystemService = require('./filesystemService');
const githubService = require('./githubService');
const crypto = require('crypto');

/**
 * Background synchronization engine for keeping local and GitHub in sync
 */
class SyncEngine extends EventEmitter {
    constructor(options = {}) {
        super();
        this.watchers = new Map();
        this.syncInterval = options.interval || 300000; // 5 minutes default
        this.intervals = new Map();
        this.rules = []; // Sync rules
        this.conflictResolution = options.conflict || 'manual'; // 'local', 'github', 'manual'
        this.cache = new Map(); // File hash cache
    }

    /**
     * Add sync rule
     * @param {Object} rule - Sync configuration
     * @param {string} rule.localPath - Local directory to sync
     * @param {string} rule.githubPath - GitHub directory to sync
     * @param {string} rule.mode - 'bidirectional', 'upload', 'download'
     * @param {boolean} rule.auto - Enable auto-sync
     */
    addRule(rule) {
        const ruleId = `rule_${Date.now()}`;
        const fullRule = {
            id: ruleId,
            ...rule,
            lastSync: null,
            status: 'active'
        };
        
        this.rules.push(fullRule);
        
        if (rule.auto) {
            this.startAutoSync(ruleId);
        }
        
        this.emit('ruleAdded', fullRule);
        return ruleId;
    }

    /**
     * Start auto-sync for rule
     */
    startAutoSync(ruleId) {
        const rule = this.rules.find(r => r.id === ruleId);
        if (!rule) return;

        // Immediate first sync
        this.performSync(ruleId);

        // Set interval
        const intervalId = setInterval(() => {
            this.performSync(ruleId);
        }, this.syncInterval);

        this.intervals.set(ruleId, intervalId);
    }

    /**
     * Stop auto-sync
     */
    stopAutoSync(ruleId) {
        if (this.intervals.has(ruleId)) {
            clearInterval(this.intervals.get(ruleId));
            this.intervals.delete(ruleId);
        }
    }

    /**
     * Perform single sync operation
     */
    async performSync(ruleId) {
        const rule = this.rules.find(r => r.id === ruleId);
        if (!rule || rule.status === 'syncing') return;

        rule.status = 'syncing';
        this.emit('syncStarted', rule);

        try {
            const changes = await this.detectChanges(rule);
            const conflicts = changes.filter(c => c.conflict);
            
            if (conflicts.length > 0 && this.conflictResolution === 'manual') {
                this.emit('conflictsDetected', rule, conflicts);
                rule.status = 'conflict';
                return;
            }

            await this.applyChanges(rule, changes);
            rule.lastSync = new Date();
            rule.status = 'active';
            
            this.emit('syncCompleted', rule, changes);

        } catch (error) {
            rule.status = 'error';
            this.emit('syncError', rule, error);
        }
    }

    /**
     * Detect changes between local and GitHub
     */
    async detectChanges(rule) {
        const changes = [];
        
        // Scan local files
        const localFiles = await this.scanLocal(rule.localPath);
        
        // Scan GitHub files
        const githubFiles = await this.scanGithub(rule.githubPath);
        
        // Create lookup maps
        const localMap = new Map(localFiles.map(f => [f.relativePath, f]));
        const githubMap = new Map(githubFiles.map(f => [f.relativePath, f]));
        
        // Check all local files
        for (const [path, localFile] of localMap) {
            const githubFile = githubMap.get(path);
            
            if (!githubFile) {
                // File exists locally only
                if (rule.mode !== 'download') {
                    changes.push({
                        type: 'upload',
                        path: path,
                        localFile,
                        reason: 'new_local'
                    });
                }
            } else {
                // File exists in both - check for modifications
                const localHash = await this.getFileHash(localFile.fullPath, 'local');
                const githubHash = await this.getFileHash(githubFile.path, 'github');
                
                if (localHash !== githubHash) {
                    const change = {
                        type: 'conflict',
                        path: path,
                        localFile,
                        githubFile,
                        localHash,
                        githubHash
                    };
                    
                    // Auto-resolve if configured
                    if (this.conflictResolution === 'local') {
                        change.type = 'upload';
                        change.resolved = 'local';
                    } else if (this.conflictResolution === 'github') {
                        change.type = 'download';
                        change.resolved = 'github';
                    }
                    
                    changes.push(change);
                }
            }
        }
        
        // Check for files only on GitHub
        for (const [path, githubFile] of githubMap) {
            if (!localMap.has(path) && rule.mode !== 'upload') {
                changes.push({
                    type: 'download',
                    path: path,
                    githubFile,
                    reason: 'new_github'
                });
            }
        }
        
        return changes;
    }

    /**
     * Apply detected changes
     */
    async applyChanges(rule, changes) {
        const transferService = require('./transferService');
        
        for (const change of changes) {
            try {
                switch (change.type) {
                    case 'upload':
                        await this.uploadFile(change.localFile, rule.githubPath);
                        break;
                    case 'download':
                        await this.downloadFile(change.githubFile, rule.localPath);
                        break;
                    case 'conflict':
                        // Skip conflicts in auto mode
                        break;
                }
                this.emit('changeApplied', rule, change);
            } catch (error) {
                this.emit('changeError', rule, change, error);
            }
        }
    }

    /**
     * Scan local directory recursively
     */
    async scanLocal(dirPath, basePath = '') {
        const files = [];
        const entries = await filesystemService.listDirectory(dirPath);
        
        for (const entry of entries) {
            const fullPath = require('path').join(dirPath, entry.name);
            const relativePath = basePath ? `${basePath}/${entry.name}` : entry.name;
            
            if (entry.type === 'directory') {
                const subFiles = await this.scanLocal(fullPath, relativePath);
                files.push(...subFiles);
            } else {
                files.push({
                    name: entry.name,
                    fullPath,
                    relativePath,
                    size: entry.size,
                    modified: entry.modified
                });
            }
        }
        
        return files;
    }

    /**
     * Scan GitHub directory recursively
     */
    async scanGithub(path) {
        const tree = await githubService.getTreeRecursive(path || '');
        return tree
            .filter(item => item.type === 'blob')
            .map(item => ({
                path: item.path,
                relativePath: path ? item.path.replace(`${path}/`, '') : item.path,
                sha: item.sha,
                size: item.size
            }));
    }

    /**
     * Get file hash for comparison
     */
    async getFileHash(path, source) {
        const cacheKey = `${source}:${path}`;
        
        // Check cache first
        if (this.cache.has(cacheKey)) {
            return this.cache.get(cacheKey);
        }
        
        let hash;
        if (source === 'local') {
            const content = await filesystemService.readFile(path);
            hash = crypto.createHash('md5').update(content).digest('hex');
        } else {
            // For GitHub, use SHA if available, otherwise fetch and hash
            try {
                const fileInfo = await githubService.getFileInfo(path);
                hash = fileInfo.sha;
            } catch (e) {
                const content = await githubService.getFileContent(path);
                hash = crypto.createHash('md5').update(content).digest('hex');
            }
        }
        
        this.cache.set(cacheKey, hash);
        return hash;
    }

    /**
     * Upload single file
     */
    async uploadFile(localFile, githubBasePath) {
        const content = await filesystemService.readFile(localFile.fullPath);
        const githubPath = `${githubBasePath}/${localFile.relativePath}`.replace(/\\/g, '/');
        
        try {
            const existing = await githubService.getFileInfo(githubPath);
            await githubService.updateFile(githubPath, content, existing.sha, `Sync ${localFile.name}`);
        } catch (e) {
            await githubService.createFile(githubPath, content, `Sync ${localFile.name}`);
        }
    }

    /**
     * Download single file
     */
    async downloadFile(githubFile, localBasePath) {
        const content = await githubService.getFileContent(githubFile.path);
        const localPath = require('path').join(localBasePath, githubFile.relativePath);
        
        // Ensure directory exists
        const dir = require('path').dirname(localPath);
        await filesystemService.createDirectory(dir);
        
        await filesystemService.writeFile(localPath, content);
    }

    /**
     * Get sync status
     */
    getStatus() {
        return {
            rules: this.rules.map(r => ({
                ...r,
                autoSync: this.intervals.has(r.id)
            })),
            activeSyncs: Array.from(this.intervals.keys())
        };
    }

    /**
     * Remove sync rule
     */
    removeRule(ruleId) {
        this.stopAutoSync(ruleId);
        this.rules = this.rules.filter(r => r.id !== ruleId);
        this.emit('ruleRemoved', ruleId);
    }
}

module.exports = SyncEngine;
```

## MODULE 9: Error Recovery System

```javascript
// server/utils/resilience.js
const fs = require('fs').promises;
const path = require('path');

/**
 * Resilience utilities for handling failures and recovery
 */
class ResilienceManager {
    constructor(options = {}) {
        this.checkpointDir = options.checkpointDir || './checkpoints';
        this.maxRetries = options.maxRetries || 3;
        this.retryDelay = options.retryDelay || 1000;
        this.checkpoints = new Map();
    }

    /**
     * Execute function with retry logic
     */
    async withRetry(operation, context = {}) {
        let lastError;
        
        for (let attempt = 1; attempt <= this.maxRetries; attempt++) {
            try {
                return await operation();
            } catch (error) {
                lastError = error;
                
                // Check if error is retryable
                if (!this.isRetryableError(error)) {
                    throw error;
                }
                
                if (attempt < this.maxRetries) {
                    const delay = this.retryDelay * Math.pow(2, attempt - 1);
                    console.log(`[Retry ${attempt}/${this.maxRetries}] ${context.name || 'operation'} failed, retrying in ${delay}ms...`);
                    await this.sleep(delay);
                }
            }
        }
        
        throw lastError;
    }

    /**
     * Check if error should be retried
     */
    isRetryableError(error) {
        // Network errors, rate limits, temporary failures
        const retryableCodes = ['ECONNRESET', 'ETIMEDOUT', 'ECONNREFUSED', 'EPIPE'];
        const retryableStatuses = [408, 429, 500, 502, 503, 504];
        
        if (error.code && retryableCodes.includes(error.code)) return true;
        if (error.status && retryableStatuses.includes(error.status)) return true;
        
        return false;
    }

    /**
     * Create checkpoint for resumable operation
     */
    async createCheckpoint(id, state) {
        const checkpointPath = path.join(this.checkpointDir, `${id}.json`);
        
        try {
            await fs.mkdir(this.checkpointDir, { recursive: true });
            await fs.writeFile(checkpointPath, JSON.stringify({
                ...state,
                timestamp: Date.now(),
                version: 1
            }));
            
            this.checkpoints.set(id, state);
            return true;
        } catch (error) {
            console.error('Failed to create checkpoint:', error);
            return false;
        }
    }

    /**
     * Load checkpoint
     */
    async loadCheckpoint(id) {
        // Check memory first
        if (this.checkpoints.has(id)) {
            return this.checkpoints.get(id);
        }
        
        // Load from disk
        const checkpointPath = path.join(this.checkpointDir, `${id}.json`);
        
        try {
            const data = await fs.readFile(checkpointPath, 'utf-8');
            const state = JSON.parse(data);
            this.checkpoints.set(id, state);
            return state;
        } catch (error) {
            return null;
        }
    }

    /**
     * Clear checkpoint
     */
    async clearCheckpoint(id) {
        this.checkpoints.delete(id);
        
        const checkpointPath = path.join(this.checkpointDir, `${id}.json`);
        try {
            await fs.unlink(checkpointPath);
        } catch (error) {
            // Ignore if doesn't exist
        }
    }

    /**
     * Resume transfer from checkpoint
     */
    async resumeTransfer(checkpointId, transferFunction) {
        const checkpoint = await this.loadCheckpoint(checkpointId);
        
        if (!checkpoint) {
            throw new Error('No checkpoint found for transfer');
        }
        
        console.log(`[Resume] Resuming transfer from checkpoint: ${checkpointId}`);
        
        // Update checkpoint with resume attempt
        checkpoint.resumeCount = (checkpoint.resumeCount || 0) + 1;
        checkpoint.lastResume = Date.now();
        
        await this.createCheckpoint(checkpointId, checkpoint);
        
        // Execute transfer with checkpoint context
        return await transferFunction(checkpoint);
    }

    /**
     * Circuit breaker pattern for failing services
     */
    createCircuitBreaker(options = {}) {
        const threshold = options.failureThreshold || 5;
        const timeout = options.resetTimeout || 30000;
        
        let failures = 0;
        let lastFailure = null;
        let state = 'CLOSED'; // CLOSED, OPEN, HALF_OPEN
        
        return async (operation) => {
            if (state === 'OPEN') {
                if (Date.now() - lastFailure > timeout) {
                    state = 'HALF_OPEN';
                } else {
                    throw new Error('Circuit breaker is OPEN');
                }
            }
            
            try {
                const result = await operation();
                
                if (state === 'HALF_OPEN') {
                    state = 'CLOSED';
                    failures = 0;
                }
                
                return result;
            } catch (error) {
                failures++;
                lastFailure = Date.now();
                
                if (failures >= threshold) {
                    state = 'OPEN';
                }
                
                throw error;
            }
        };
    }

    sleep(ms) {
        return new Promise(resolve => setTimeout(resolve, ms));
    }
}

module.exports = new ResilienceManager();
```

```javascript
// server/services/cacheService.js
const NodeCache = require('node-cache');

/**
 * Caching service for file trees and metadata
 * Reduces API calls and improves performance for large repositories
 */
class CacheService {
    constructor() {
        // Tree cache - longer TTL for directory structures
        this.treeCache = new NodeCache({
            stdTTL: 300, // 5 minutes
            checkperiod: 60,
            useClones: false
        });
        
        // File content cache - shorter TTL
        this.contentCache = new NodeCache({
            stdTTL: 60, // 1 minute
            checkperiod: 30,
            maxKeys: 1000 // Limit memory usage
        });
        
        // Metadata cache
        this.metaCache = new NodeCache({
            stdTTL: 120, // 2 minutes
            checkperiod: 60
        });
        
        this.stats = {
            hits: 0,
            misses: 0,
            evictions: 0
        };
        
        this.setupEventHandlers();
    }

    setupEventHandlers() {
        this.treeCache.on('expired', (key, value) => {
            this.stats.evictions++;
        });
    }

    /**
     * Get cached tree
     */
    getTree(key) {
        const cached = this.treeCache.get(key);
        if (cached) {
            this.stats.hits++;
            return cached;
        }
        this.stats.misses++;
        return null;
    }

    /**
     * Set cached tree
     */
    setTree(key, tree, ttl = 300) {
        return this.treeCache.set(key, tree, ttl);
    }

    /**
     * Get cached file content
     */
    getContent(key) {
        const cached = this.contentCache.get(key);
        if (cached) {
            this.stats.hits++;
            return cached;
        }
        this.stats.misses++;
        return null;
    }

    /**
     * Set cached content
     */
    setContent(key, content, ttl = 60) {
        // Don't cache large files (>1MB)
        if (Buffer.byteLength(content, 'utf8') > 1024 * 1024) {
            return false;
        }
        return this.contentCache.set(key, content, ttl);
    }

    /**
     * Get cached metadata
     */
    getMeta(key) {
        return this.metaCache.get(key);
    }

    /**
     * Set cached metadata
     */
    setMeta(key, meta, ttl = 120) {
        return this.metaCache.set(key, meta, ttl);
    }

    /**
     * Invalidate cache entries by pattern
     */
    invalidate(pattern) {
        const treeKeys = this.treeCache.keys();
        const contentKeys = this.contentCache.keys();
        const metaKeys = this.metaCache.keys();
        
        const regex = new RegExp(pattern);
        
        let count = 0;
        
        treeKeys.forEach(key => {
            if (regex.test(key)) {
                this.treeCache.del(key);
                count++;
            }
        });
        
        contentKeys.forEach(key => {
            if (regex.test(key)) {
                this.contentCache.del(key);
                count++;
            }
        });
        
        metaKeys.forEach(key => {
            if (regex.test(key)) {
                this.metaCache.del(key);
                count++;
            }
        });
        
        return count;
    }

    /**
     * Invalidate specific path
     */
    invalidatePath(path) {
        // Invalidate exact path and parent paths
        this.treeCache.del(`tree:${path}`);
        this.contentCache.del(`content:${path}`);
        this.metaCache.del(`meta:${path}`);
        
        // Invalidate parent trees
        const parts = path.split('/');
        while (parts.length > 0) {
            parts.pop();
            const parentPath = parts.join('/');
            this.treeCache.del(`tree:${parentPath}`);
        }
    }

    /**
     * Get cache statistics
     */
    getStats() {
        return {
            ...this.stats,
            tree: {
                keys: this.treeCache.keys().length,
                vsize: this.treeCache.getStats().vsize
            },
            content: {
                keys: this.contentCache.keys().length,
                vsize: this.contentCache.getStats().vsize
            },
            meta: {
                keys: this.metaCache.keys().length,
                vsize: this.metaCache.getStats().vsize
            }
        };
    }

    /**
     * Clear all caches
     */
    clear() {
        this.treeCache.flushAll();
        this.contentCache.flushAll();
        this.metaCache.flushAll();
        this.stats = { hits: 0, misses: 0, evictions: 0 };
    }

    /**
     * Wrap async function with caching
     */
    wrap(fn, cacheKey, type = 'content', ttl) {
        return async (...args) => {
            const key = typeof cacheKey === 'function' ? cacheKey(...args) : cacheKey;
            
            const cache = type === 'tree' ? this.treeCache : 
                         type === 'meta' ? this.metaCache : this.contentCache;
            
            const cached = cache.get(key);
            if (cached !== undefined) {
                this.stats.hits++;
                return cached;
            }
            
            this.stats.misses++;
            const result = await fn(...args);
            
            if (result !== null && result !== undefined) {
                cache.set(key, result, ttl);
            }
            
            return result;
        };
    }
}

module.exports = new CacheService();
```

```javascript
    });
});

// Queue management endpoints
router.post('/queue', async (req, res) => {
    try {
        const jobs = req.body.jobs;
        if (!Array.isArray(jobs)) {
            return res.status(400).json({
                success: false,
                error: 'jobs array required'
            });
        }

        const jobIds = transferQueue.addBatch(jobs);
        
        res.json({
            success: true,
            jobIds,
            queueStatus: transferQueue.getStatus()
        });
    } catch (error) {
        res.status(500).json({
            success: false,
            error: error.message
        });
    }
});

router.get('/queue/status', (req, res) => {
    res.json({
        success: true,
        ...transferQueue.getStatus()
    });
});

router.post('/queue/pause', (req, res) => {
    transferQueue.pause();
    res.json({ success: true, message: 'Queue paused' });
});

router.post('/queue/resume', (req, res) => {
    transferQueue.resume();
    res.json({ success: true, message: 'Queue resumed' });
});
```

```javascript
async function openDiffViewer(panel, file) {
    const diffContainer = document.createElement('div');
    diffContainer.style.cssText = 'position:fixed;top:0;left:0;right:0;bottom:0;z-index:100;background:#1f2937;';
    document.body.appendChild(diffContainer);
    
    const diffViewer = new DiffViewer();
    diffViewer.init(diffContainer);
    
    if (panel === 'local') {
        const localPath = `${state.local.currentPath}/${file.name}`;
        const githubPath = `${state.github.currentPath}/${file.name}`.replace(/^\/+/, '');
        await diffViewer.compareLocalWithGithub(localPath, githubPath, file.name);
    } else {
        // Reverse comparison
        const githubPath = file.path || file.name;
        const localPath = `${state.local.currentPath}/${file.name}`;
        await diffViewer.compareLocalWithGithub(localPath, githubPath, file.name);
    }
    
    // Add close button
    const closeBtn = document.createElement('button');
    closeBtn.innerHTML = '<i class="fas fa-times"></i> Close Diff';
    closeBtn.style.cssText = 'position:absolute;top:10px;right:10px;padding:8px 16px;background:#374151;color:white;border:none;border-radius:4px;cursor:pointer;';
    closeBtn.onclick = () => {
        diffViewer.destroy();
        diffContainer.remove();
    };
    diffContainer.appendChild(closeBtn);
}
```
